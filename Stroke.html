from sklearn.model_selection import cross_val_score
from sklearn.neighbors import KNeighborsClassifier
kfold_accuracy=[]
for i in range(1,30,2):
    KNN_KFold = KNeighborsClassifier(i)
    KFold = cross_val_score(KNN_KFold,data_X,data_Y, cv=5)
    kfold_accuracy.append(KFold.mean())

plt.plot(range(1,30,2),kfold_accuracy,label='KFold')
plt.xticks(range(1,30,2))
plt.xlabel('neighbors')

plt.grid()

由上圖可知，n 約在 13 之後就趨於平緩¶

- 設定 n = 13 並計算模型準確率¶

In [26]:
from sklearn.model_selection import KFold
KNN = KNeighborsClassifier(13)
KF_CV = KFold(n_splits=5, shuffle=True)
accuracy=[]
for train, test in KF_CV.split(data_X):
    KNN.fit(data_X[train],data_Y[train])
    Prediction = KNN.predict(data_X[test])
    accuracy.append(KNN.score(data_X[test],data_Y[test]))
print('KNN:',np.mean(accuracy)*100,'%')
